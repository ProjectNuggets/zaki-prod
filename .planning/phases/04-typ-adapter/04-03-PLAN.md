5. Refactor requireWorkspaceAccess — extract novaUserId from zakiUser, call updated workspaceVisibleForSession:
```javascript
async function requireWorkspaceAccess(req, res) {
  const authResult = await requireAuthUser(req, res);
  if (!authResult) return null;

  const { zakiUser, email } = authResult;
  if (!zakiUser.verified) {
    res.status(403).json({ error: "Email is not verified." });
    return null;
  }

  const slug = String(req.params.slug || "").trim().toLowerCase();
  if (!slug) {
    res.status(400).json({ error: "Workspace slug is required." });
    return null;
  }

  if (!zakiUser.nova_user_id) {
    res.status(403).json({ error: "Workspace access requires a linked TYP account." });
    return null;
  }

  const accessCheck = await workspaceVisibleForSession(zakiUser.nova_user_id, slug);
  if (!accessCheck.success) {
    res.status(accessCheck.status || 502).json({
      error: accessCheck.error || "Unable to verify workspace access.",
    });
    return null;
  }
  if (!accessCheck.visible) {
    res.status(403).json({ error: "You do not have access to this workspace." });
    return null;
  }

  return {
    authResult,
    email,
    zakiUser,
    slug,
  };
}
```

```javascript
const listWorkspacesHandler = async (req, res) => {
  try {
    const authResult = await requireAuthUser(req, res);
    if (!authResult) return;
    const { zakiUser } = authResult;

    if (!zakiUser.nova_user_id) {
      res.status(403).json({
        success: false,
        error: "Workspace listing requires a linked TYP account.",
      });
      return;
    }

    const upstream = await fetchTypWorkspaces(zakiUser.nova_user_id);
    const data = await upstream.json().catch(() => ({}));
    if (!upstream.ok || !Array.isArray(data?.workspaces)) {
      res.status(upstream.status || 502).json({
        success: false,
        error: data?.error || data?.message || "Unable to fetch workspaces.",
      });
      return;
    }

    const hiddenSlugs = await listHiddenWorkspaceSlugsForUser(zakiUser.id);
    const filtered = data.workspaces.filter((workspace) => {
      const slug = String(workspace?.slug || "").trim().toLowerCase();
      return slug && (hiddenSlugs.size === 0 || !hiddenSlugs.has(slug));
    });
    const metadataBySlug = await listWorkspaceMetadata(
      filtered.map((workspace) => workspace?.slug)
    );
    res.status(200).json({
      ...data,
      workspaces: filtered.map((workspace) =>
        mergeWorkspaceMetadata(
          workspace,
          metadataBySlug.get(normalizeWorkspaceSlugValue(workspace?.slug))
        )
      ),
    });
  } catch (error) {
    console.error("[Workspace] listWorkspacesHandler error:", error);
    res.status(500).json({ success: false, error: "Failed to load workspaces." });
  }
};
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client -> ZAKI backend (/workspaces, /workspace/:slug, stream-chat) | Untrusted: Authorization header is a ZAKI JWT, validated by requireAuthUser |
| ZAKI backend -> TYP pod | Trusted server-to-server; admin API key; nova_user_id sourced from DB not client |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-07 | Elevation of Privilege | requireWorkspaceAccess nova_user_id | mitigate | nova_user_id read from zakiUser (DB column, populated by requireAuthUser); client cannot supply it. Guard added for null nova_user_id returns 403. |
| T-04-08 | Spoofing | listWorkspacesHandler / deleteWorkspaceHandler | mitigate | requireAuthUser validates ZAKI JWT locally (Phase 2) or TYP session; zakiUser.nova_user_id comes from DB row matched on JWT sub. Client cannot spoof a different user's nova_user_id. |
| T-04-09 | Information Disclosure | streamChatHandler | mitigate | authHeader (user's ZAKI JWT) no longer forwarded to TYP; TYP receives admin key only. User can no longer probe TYP by sending a crafted auth header through the ZAKI stream endpoint. |
| T-04-10 | Denial of Service | deleteWorkspaceHandler verifyWorkspaceDeleted | accept | Poll loop (3 attempts, 150ms gap) unchanged; admin API key means TYP throttle applies per-server not per-user |
</threat_model>
