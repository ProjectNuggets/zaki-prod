```typescript
// Types
export interface BrainGraphResponse { nodes, edges, trimmed, total_skipped, total_nodes_in_corpus, semantic_degraded }
export interface BrainGraphNode { id, kind, created_at, session_id, summary, valid_to }
export type BrainGraphEdge = session | semantic | reference discriminated union
export interface BrainTimelineResponse { entries, next_cursor, has_more }
export interface BrainTimelineEntry { id, key, kind, created_at, session_id, summary, valid_to }
export interface BrainComposeRequest { title, content, references, category?, key? }
export interface BrainComposeResponse { key, synthesized_by, references_count, category, composed_at }

// Fetch wrappers (all use apiRequest())
export async function fetchBrainGraph(userId, opts?) -> Promise<BrainGraphResponse>
export async function fetchBrainTimeline(userId, opts?) -> Promise<BrainTimelineResponse>
export async function postBrainCompose(userId, body) -> Promise<BrainComposeResponse>
```
